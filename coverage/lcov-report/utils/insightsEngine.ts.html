    checkInCount: weekWindow.length,

    avgStress: parseFloat(mean(weekWindow.map(c => stressToNum(c.stressLevel))).toFixed(1)),
    moodTrend,
    energyTrend,
    stressTrend,
    periodLabel,
  };
}
